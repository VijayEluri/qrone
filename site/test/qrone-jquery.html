make jQuery Plugin.

java -jar qrone-x.y.z.jar -l js parts.html -o parts.js

include it

<script src="jquery.min.js"></script>
<script src="parts.js"></script>

use it

jQuery("#someid").qrone('parts');
// insert contents of parts.html into #someid element.

jQuery(".cssclass").append(qrone('parts'));
// append contents of parts.html into all element with 'cssclass' class.
qrone('parts').appendTo(jQuery(".class"));
// return value of qrone() is also a jQuery object.
require('CLASS');

jQuery(".class").qrone('CLASS');

jQuery(".class").append(qrone('CLASS'));
